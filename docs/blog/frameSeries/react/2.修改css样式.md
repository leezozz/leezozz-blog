### 1.结合 styled-components 修改 antd 组件CSS样式
```jsx
// 首先需要安装
yarn add styled-components

// 首先新建一个tsx文件存放样式 NavTabs.styled.tsx
// NavTabs: 注意是大写
// 设置tabs内容不显示
export const NavTabs = styled.div`
  padding: 0 16px;

  .ant-tabs {  
    .ant-tabs-content-holder {
      display: none;
    }
  }
`;

// Nav.tsx
// NavTabs作为标签使用
import { NavTabs } from "./NavTabs.styled.tsx";
<NavTabs>
   <Tabs defaultActiveKey="1" items={items} onChange={onChange} />
</NavTabs>
```

### 2.在组件中引入 [name].module.scss 文件
```jsx
// 首先创建对应文件的scss文件 Register.module.scss
.container {
  height: calc(100vh - 64px - 65px);
  display: flex;
  justify-content: center;
  align-items: center;a
  flex-direction: column;
  background-color: #fff;
 .content {
	  box-shadow: 0 4px 10px gray;
	  padding: 32px 52px;
	  border-radius: 8px;
 }
}

// 对应注册文件：Register.tsx
import styles from './Register.module.scss'
<div className={styles.container}>
  <div className={styles.content}>
	  注册内容xxx
	</div>
</div>
```

      border: 'none',
